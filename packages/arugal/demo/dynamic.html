How It Works

// Initialize on an empty container
const manager = Arugal();
const gallery = manager.getInstance(document.getElementById('dynamic-gallery'));

// Add images dynamically
gallery.addItems([
    { src: 'image1.jpg', title: 'First Image' },
    { src: 'image2.jpg', title: 'Second Image', description: 'A description' }
]);

// Replace all images
gallery.setItems([
    { src: 'new1.jpg', title: 'New Image 1' },
    { src: 'new2.jpg', title: 'New Image 2' }
]);

// Remove specific image
gallery.removeItem(0);  // Remove first image

// Clear all images
gallery.clear();

// Navigation
gallery.next();
gallery.prev();
gallery.goTo(2);

// Open lightbox programmatically
gallery.openLightbox();
gallery.openLightbox(1);  // Open at specific index
